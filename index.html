function generatePDF() { // UI থেকে data নেওয়া document.getElementById("pdfOrderDate").innerText = document.getElementById("orderDate").value; document.getElementById("pdfDueDate").innerText = document.getElementById("dueDate").value; document.getElementById("pdfDistributor").innerText = document.querySelectorAll("input")[2].value; document.getElementById("pdfAddress").innerText = document.querySelectorAll("input")[3].value; document.getElementById("pdfSeason").innerText = document.querySelectorAll("input")[4].value; document.getElementById("pdfType").innerText = document.querySelectorAll("input")[5].value; document.getElementById("pdfSlab").innerText = document.querySelectorAll("input")[6].value; document.getElementById("pdfApproval").innerText = document.querySelectorAll("input")[7].value; document.getElementById("pdfOfficer").innerText = document.querySelectorAll("input")[8].value; document.getElementById("pdfMobile").innerText = document.querySelectorAll("input")[9].value; document.getElementById("pdfTerritory").innerText = document.querySelectorAll("input")[10].value; // product data let table = document.getElementById("productTable"); let pdfProducts = document.getElementById("pdfProducts"); pdfProducts.innerHTML = ""; for (let i = 1; i < table.rows.length; i++) { let cells = table.rows[i].cells; let row = ` ${cells[0].children[0].value} ${cells[1].children[0].value} ${cells[2].children[0].value} ${cells[3].children[0].value} ${cells[4].innerText} `; pdfProducts.innerHTML += row; } document.getElementById("pdfTotalQty").innerText = document.getElementById("totalQty").innerText; document.getElementById("pdfTotalValue").innerText = document.getElementById("totalValue").innerText; // PDF render let pdfArea = document.getElementById("pdf-format"); pdfArea.style.display = "block"; html2canvas(pdfArea).then(canvas => { const imgData = canvas.toDataURL("image/png"); const { jsPDF } = window.jspdf; const pdf = new jsPDF(); pdf.addImage(imgData, 'PNG', 10, 10, 190, 0); pdf.save("order.pdf"); pdfArea.style.display = "none"; }); }
